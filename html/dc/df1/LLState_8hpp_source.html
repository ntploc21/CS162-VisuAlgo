314template< typename T >
315void State::LLState< T >::AddIntFieldOperationOption(
316 GUIComponent::OperationContainer::Ptr container, std::string title,
317 Core::Deque< IntegerInput > fields,
318 std::function< void(std::map< std::string, std::string >) > action) {
319 GUIComponent::OptionInputField::Ptr button(
320 new GUIComponent::OptionInputField(GetContext().fonts));
321 Core::Deque< GUIComponent::InputField::Ptr > intFields;
322 for (auto field : fields) {
323 GUIComponent::IntegerInputField::Ptr intField(
324 new GUIComponent::IntegerInputField(GetContext().fonts));
325 intField.get()->SetLabel(field.label);
326 intField.get()->SetInputFieldSize((Vector2){(float)field.width, 30});
327 intField.get()->SetConstraint(field.minValue, field.maxValue);
328 intField.get()->Randomize();
329 intFields.push_back(intField);
330 }
331
332 button.get()->SetOption(title, intFields, action);
333
334 container.get()->pack(button);
335}
