93 int nextIndex;

96 (arrowStateNext[nextIndex] != ArrowType::Skip &&
97 arrowStatePrev[nextIndex] != ArrowType::Skip))
98 break;
99
100 if (nextIndex == list.size())
101 break; // all of the nodes behind the current node is hidden, so we
102 // don't need to draw any arrow

106 end.x += base.x, end.y += base.y;
