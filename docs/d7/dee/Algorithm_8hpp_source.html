234template< typename GUIAlgorithm, typename AnimationState >
235std::vector< int >
236Algorithm::Algorithm< GUIAlgorithm, AnimationState >::UserDefinedGenerator(
237 std::string input) {
238 bool canParse =
239 (!input.empty() &&
240 std::all_of(
241 input.begin(), input.end(),
242 [](char c) { return ('0' <= c && c <= '9') || (c == ','); }) &&
243 input.back() != ',');
244 if (!canParse) return EmptyGenerator();
245
246 std::vector< int > answer;
247
248 std::istringstream f(input);
249 std::string s;
250 while (std::getline(f, s, ',')) {
251 try {
252 answer.push_back(atoi(s.c_str()));
253 } catch (char* e) {
254 }
255 }
256 return answer;
257}

268inline void
269Algorithm::Algorithm< GUIAlgorithm, AnimationState >::GenerateRelayoutAnimation(
270 Vector2 newPosition) {
271 AnimationState animRelayout =
272 GenerateAnimation(0.5, -1,
273 "Re-layout the Linked List for visualization "
274 "(not in the actual Linked "
275 "List).\nThe whole process is still O(1).");
276 animRelayout.SetAnimation(
277 [this, newPosition](GUIAlgorithm srcDS, float playingAt, Vector2 base) {
278 Vector2 newPos = srcDS.GetPosition();
279 newPos.x += (newPosition.x - newPos.x) * playingAt;
280 newPos.y += (newPosition.y - newPos.y) * playingAt;
281 srcDS.SetPosition(newPos);
282
283 srcDS.Draw(base, playingAt);
284
285 return srcDS;
286 });
287 animController->AddAnimation(animRelayout);
288}

342template< typename GUIAlgorithm, typename AnimationState >
343void Algorithm::Algorithm< GUIAlgorithm, AnimationState >::ApplyInput(
344 std::vector< int > input, std::size_t nMaxSize) {
345 if (input.size() > nMaxSize) input.resize(nMaxSize);
346 InitAction({});
347
348 codeHighlighter->SetShowCode(false);
349 codeHighlighter->SetShowAction(false);
350
351 visualizer.Import(input);
352
353 AnimationState state;
354 state.SetDuration(0.5);
355 state.SetHighlightLine(-1);
356 state.SetSourceDataStructure(visualizer);
357 state.SetAnimation([this](GUIAlgorithm srcDS, float playingAt,
358 Vector2 base) {
359 auto& nodes = srcDS.GetList();
360 for (GUIVisualizer::Node& node : nodes) {
361 node.SetRadius(AnimationFactory::ElasticOut(playingAt) * 20);
362 node.SetValueFontSize(AnimationFactory::ElasticOut(playingAt) * 24);
363 node.SetLabelFontSize(AnimationFactory::ElasticOut(playingAt) * 20);
364 }
365 srcDS.Draw(base, playingAt, true);
366 return srcDS;
367 });
368
369 animController->AddAnimation(state);
370 animController->Reset();
371 animController->InteractionLock();
372}
