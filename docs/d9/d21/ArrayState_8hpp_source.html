325template< typename T >
326void State::ArrayState< T >::AddIntFieldOperationOption(
327 GUIComponent::OperationContainer::Ptr container, std::string title,
328 Core::Deque< IntegerInput > fields,
329 std::function< void(std::map< std::string, std::string >) > action) {
330 GUIComponent::OptionInputField::Ptr button(
331 new GUIComponent::OptionInputField(GetContext().fonts));
332 Core::Deque< GUIComponent::InputField::Ptr > intFields;
333 for (auto field : fields) {
334 GUIComponent::IntegerInputField::Ptr intField(
335 new GUIComponent::IntegerInputField(GetContext().fonts));
336 intField.get()->SetLabel(field.label);
337 intField.get()->SetInputFieldSize((Vector2){(float)field.width, 30});
338 intField.get()->SetConstraint(field.minValue, field.maxValue);
339 intField.get()->Randomize();
340 intFields.push_back(intField);
341 }
342
343 button.get()->SetOption(title, intFields, action);
344
345 container.get()->pack(button);
346}
